## c# 중급세션 과제
+ 문제 링크: https://www.acmicpc.net/problem/9252   
DP의 대표문제 LCS를 풀어보았읍니다   
튜플은 개사기 자료형이네요   
```c#
using System;

namespace LCS{
    public enum direction: byte{  // 배열 b를 해당 열거형으로 만듦
        up,
        left,
        diagonal
    }
    public class LCS{
        public static void Main(string[] args){
            string x = " " + Console.ReadLine().Replace(" ", string.Empty);
            string y = " " + Console.ReadLine().Replace(" ", string.Empty);
            int m = x.Length;  // x 길이 + 1 (앞에 공백을 넣어 길이를 1 늘림)
            int n = y.Length;  // y 길이 + 1
            (int[,] c, direction[,] b) = LCS_LENGTH(x,y,m,n);
            Console.WriteLine(c[m-1,n-1]);
            PRINT_LCS(b,x,m-1,n-1);
        }
        public static (int[,], direction[,]) LCS_LENGTH(string x, string y, int m, int n)
        {
            int[,] c = new int[m,n];              // 레퍼런스 형식, 배열 자체는 힙 영역에 할당
            direction[,] b = new direction[m,n];  

            int i,j;
            for(i=1; i<m; i++) c[i,0] = 0;
            for(j=0; j<m; j++) c[j,0] = 0;

            for(i=1; i<m; i++){
                for(j=1; j<n; j++){
                    if(x[i] == y[j]){
                        c[i,j] = c[i-1,j-1] + 1;
                        b[i,j] = direction.diagonal;
                    }
                    else if(c[i-1,j] >= c[i,j-1]){
                        c[i,j] = c[i-1,j];
                        b[i,j] = direction.up;
                    }
                    else{
                        c[i,j] = c[i,j-1];
                        b[i,j] = direction.left;
                    }
                }
            }

            return (c,b);  // 레퍼런스를 튜플로 반환
        }

        public static void PRINT_LCS(direction[,] b, string x, int i, int j)
        {
            if(i==0 || j==0) return;
            if(b[i,j]==direction.diagonal){
                PRINT_LCS(b,x,i-1,j-1);
                Console.Write(x[i]);
            }
            else if(b[i,j]==direction.up) PRINT_LCS(b,x,i-1,j);
            else PRINT_LCS(b,x,i,j-1);
        }

    }

}
```
